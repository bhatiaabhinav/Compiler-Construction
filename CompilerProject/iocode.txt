strConcat proc
    pusha
    lea si,tempStr2
    lea di,tempStr
    mov bx,0
newIter:
    cmp [si],'$'
    je concatNext

    mov cx,[si]
    mov [di],cx
    inc si
    inc di
    inc bx
    cmp bx,21
    jne newIter
    popa
    ret 
concatNext:
    lea si,tempStr3
nxtItr:
    cmp [si],'$'
    je concatLastDollar
    cmp bx,20
    je concatLastDollar
    mov cx,[si]
    mov [di],cx
    inc si
    inc di
    inc bx
    jmp nxtItr
concatLastDollar:
    mov [di],'$'
    popa
    ret    
endp

print_reg_asm proc
cmp ax, 0
jne  print_reg_asm_intr
    push ax
    mov al, '0'
    mov ah, 0eh
    int 10h
    pop ax
    ret 
 print_reg_asm_intr:
    pusha
    mov dx, 0
    cmp ax, 0
    je  print_pop
    mov bx, 10
    div bx    
    call  print_reg_asm_intr
    mov ax, dx
    add al, 30h
    mov ah, 0eh
    int 10h    
    jmp  print_pop
 print_pop:
    popa  
    ret  
endp
